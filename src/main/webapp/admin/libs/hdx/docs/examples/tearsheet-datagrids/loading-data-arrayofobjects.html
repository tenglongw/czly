Data loading: ajax array of object

A Datagrid loading data with AJAX. Array of Objects JSON response.

Example
Usage

Markup:

<form class="form-inline pull-right">
  <div class="input-group pull-right">
     <input type="text" class="form-control input-sm" data-filter-table="dt-ajax-objects">
     <div class=input-group-btn">
         <button class="btn btn-icon"><i class="icon-search"></i></button>
     </div>
  </div>
</form>
<table cellpadding="0" cellspacing="0" border="0" class="table table-bordered" data-table-name="dt-ajax-objects"></table>

Script:

// Ajax with DataTables, array of objects
$('table[data-table-name="dt-ajax-objects"]').basicdatagrid({
  'aoColumns': [
    { 'sTitle': 'Rendering engine', 'mData': 'engine', 'sClass': 'essential' },
    { 'sTitle': 'Browser', 'mData': 'browser', 'sClass': 'essential' },
    { 'sTitle': 'Platform(s)', 'mData': 'platform', 'sClass': 'optional' },
    { 'sTitle': 'Engine version', 'mData': 'version' },
    { 'sTitle': 'CSS grade', 'mData': 'grade', 'sClass': 'essential' }
  ],
  'sAjaxSource': '../assets/data/ao.browsers.json',
  'useFloater': false,
  'isResponsive': true
});
