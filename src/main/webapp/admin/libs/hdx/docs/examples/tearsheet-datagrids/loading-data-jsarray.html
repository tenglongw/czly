Data loading: array of objects

A Datagrid loading data from a JavaScript Array.

Example
Usage

Markup:

<form class="form-inline pull-right">
  <div class="input-group pull-right">
     <input type="text" class="form-control input-sm" data-filter-table="dt-array">
     <div class=input-group-btn">
         <button class="btn btn-icon"><i class="icon-search"></i></button>
     </div>
  </div>
</form>
<table cellpadding="0" cellspacing="0" border="0" class="table table-bordered" data-table-name="dt-array"></table>

Script:

// JavaScript array and columns
$('table[data-table-name="dt-array"]').basicdatagrid({
  'aoColumns': [
    { 'sTitle': 'Schedule', 'sClass': 'essential' },
    { 'sTitle': 'Account Status', 'sClass': 'essential' },
    { 'sTitle': 'DBA Name', 'sClass': 'essential' },
    { 'sTitle': 'Customer Info', 'sClass': 'optional' }
  ],
  'aaData': [
    ['foo', 'bar', 'baz', 'bueller'],
    ['possum', 'raccoon', 'gravy', 'snowflake'],
    ['peanut', 'epsilon', 'flapjack', 'nominee']
  ],
  'useFloater': false,
  'isResponsive': true
});
