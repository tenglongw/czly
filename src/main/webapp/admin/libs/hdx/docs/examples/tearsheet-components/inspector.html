Description

An inspector is used to display detailed information about any particular cell in a data grid. It is inserted as an expanded row in the table below the relevant row. Clicking the cell a second time closes the inspector. Single or multiple inspectors per table are supported as a configurable option.

Usage

Markup:

<table data-table-name="myTable"  class="table"  >
//add table headers and rows here.
</table>

Script:

require(['jquery','inspector','modules', 'ge-bootstrap', 'datagrids' ],function ($, inspector) {
	
	$(function() {
      	var table = $('table[data-table-name="myTable"]').iidsBasicDataGrid({
        useFloater: false,
        isResponsive: true,
        bInfo: false,
        bSortClasses: false,
      	bPaginate: false,
      	fnCreatedRow: function fnCreatedRow(nRow) {
       		  $('td',nRow).addClass("inspectable");
	       	  $('td', nRow).click(function() {
	      			var inspectorContent= HTML content to display
	      			var allowMultipleInspectors = true/false;
	      			//This will be called per cell click
	      			$(this).inspector(nRow, tbl ,$(inspectorContent), allowMultipleInspectors);
	          });
        }
      });
      // fetch the (hungarian) table for later use
      var tbl = table.dataTable();
    });
  });
