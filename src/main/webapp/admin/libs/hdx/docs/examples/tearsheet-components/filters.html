Filter 001
Description

A Filter Tray holds contains the HDx Rule Builder, primarily used for filtering and highlighting tables. It is initially hidden. Bootstrap's collapse functionality is used to hide and show the tray.

The Filter Button is a split button with a dropdown. Filter tray can displayed or hidden by clicking on the filter label. Buttons for filter presets can stack alongside the filter button.

Usage

Markup:

<div class="tray filter-tray top collapse">
  <div class="tray-inner">
    <div class="filter-rule-group"></div>
    <!-- client-side template -->
    <script type="text/template" id="filter-rule-template">
      <form class="inline-form filter-rule">
        <span class="filter-item">
          <select>
            <option value="show">Show Only</option>
            <option value="hide">Hide</option>
            <option value="highlight">Highlight</option>
          </select>
        </span>
        <span class="filter-item">
          <select>
            <option value="siteName">Site Name</option>
            <option value="turbineSN">Turbine SN</option>
            <option value="average">Average</option>
            <option value="median">Median</option>
            <option value="standardDeviation">Standard Deviation</option>
            <option value="percentile">Percentile</option>
          </select>
        </span>
        <span class="filter-item">
          <select>
            <option value="=">=</option>
            <option value=">">></option>
            <option value=">=">>=</option>
            <option value="<"><</option>
            <option value="<="><=</option>
            <option value="startsWidth">starts with</option>
            <option value="endsWith">ends width</option>
            <option value="contains">contains</option>
            <option value="doesNotContain">does not contain</option>
          </select>
        </span>
        <span class="filter-item">
          <input type="text" value="sample">
        </span>
        <span class="filter-item">
          <button type="button" class="btn-link" data-toggle="filter-rule-remove"><i class="icon-remove-sign"></i></button>
        </span>
      </form>
    </script>
    <!-- end template -->
    <button class="btn btn-link pull-left btn-default btn-lg" data-toggle="filter-rule-add" type="button"><i class="icon-plus-sign"></i> Add Condition</button>
    <div class="pull-right">
      <button class="btn btn-inverse">Cancel</button>
      <button class="btn btn-primary">Apply</button>
    </div>
  </div>
</div>

Script:

require( ['filtertray'], function() {
  $('.tray').iidsFilterTray();
});
