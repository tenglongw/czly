Data loading: ajax array of arrays

A Datagrid loading data with AJAX. Array of Arrays JSON response.

Example
Usage

Markup:

<form class="form-inline pull-right">
  <div class="input-group pull-right">
     <input type="text" class="form-control input-sm" data-filter-table="dt-ajax-array">
     <div class=input-group-btn">
         <button class="btn btn-icon"><i class="icon-search"></i></button>
     </div>
  </div>
</form>
<table cellpadding="0" cellspacing="0" border="0" class="table table-bordered" data-table-name="dt-ajax-array"></table>

Script:

// Ajax with DataTables, array of arrays
$('table[data-table-name="dt-ajax-array"]').basicdatagrid({
  'aoColumns': [
    { 'sTitle': 'Rendering engine', 'sClass': 'essential' },
    { 'sTitle': 'Browser', 'sClass': 'essential' },
    { 'sTitle': 'Platform(s)', 'sClass': 'optional' },
    { 'sTitle': 'Engine version' },
    { 'sTitle': 'CSS grade', 'sClass': 'essential' }
  ],
  'sAjaxSource': '../assets/data/aa.browsers.json',
  'useFloater': false,
  'isResponsive': true
});
